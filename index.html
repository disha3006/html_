      @Transactional
      public String approveLeave(
      String managerAdid, Long leaveId) {

      User manager = getManager(managerAdid);

      LeaveRequest leave = leaveRequestRepository
      .findById(leaveId)
      .orElseThrow(() ->
      new RuntimeException(
      "Leave not found"));

      // Check belongs to manager's team
      if (!leave.getUser().getManager().getId()
      .equals(manager.getId())) {
      return "error:Unauthorized";
      }

      // Update status
      leave.setStatus(LeaveStatus.APPROVED);
      leave.setApprovedBy(manager);
      leave.setActionedAt(LocalDateTime.now());
      leaveRequestRepository.save(leave);

      // Mark attendance as ON_LEAVE
      markAttendanceAsOnLeave(leave);

      // Notify employee
      notificationService.sendNotification(
      leave.getUser(),
      "Leave Approved",
      "Your " + leave.getLeaveType().name()
      + " leave for "
      + leave.getStartDate() + " to "
      + leave.getEndDate()
      + " has been approved by "
      + manager.getFullName(),
      NotificationType.LEAVE_APPROVED);

      log.info("Leave approved: {} by {}",
      leaveId, managerAdid);

      return "success:Leave approved successfully";
      }

      // ── Mark attendance as ON_LEAVE ────────────────────────────

      private void markAttendanceAsOnLeave(
      LeaveRequest leave) {

      LocalDate current = leave.getStartDate();

      while (!current.isAfter(leave.getEndDate())) {

      if (current.getDayOfWeek()
      != DayOfWeek.SATURDAY
      && current.getDayOfWeek()
      != DayOfWeek.SUNDAY) {

      OptionalAttendance> existing =
        attendanceRepository
        .findByUserAndDate(
        leave.getUser(),
        current);

        if (existing.isPresent()) {
        existing.get().setStatus(
        AttendanceStatus.ON_LEAVE);
        attendanceRepository
        .save(existing.get());
        } else {
        Attendance attendance =
        Attendance.builder()
        .user(leave.getUser())
        .date(current)
        .status(AttendanceStatus
        .ON_LEAVE)
        .isRegularized(false)
        .isWorkedOnHoliday(false)
        .build();
        attendanceRepository
        .save(attendance);
        }
        }
        current = current.plusDays(1);
        }
        }

    // ── Approve correction ─────────────────────────────────────

    @Transactional
    public String approveCorrection(
    String managerAdid, Long correctionId) {

    User manager = getManager(managerAdid);

    Correction correction = correctionRepository
    .findById(correctionId)
    .orElseThrow(() ->
    new RuntimeException(
    "Correction not found"));

    if (!correction.getUser().getManager().getId()
    .equals(manager.getId())) {
    return "error:Unauthorized";
    }

    // Update attendance record
    Attendance attendance =
    correction.getAttendance();

    if (correction.getRequestedClockIn() != null) {
    attendance.setClockIn(
    correction.getRequestedClockIn());
    }
    if (correction.getRequestedClockOut() != null) {
    attendance.setClockOut(
    correction.getRequestedClockOut());
    }

    attendance.setIsRegularized(true);
    attendanceRepository.save(attendance);

    // Update correction
    correction.setStatus(
    CorrectionStatus.APPROVED);
    correction.setReviewedBy(manager);
    correction.setActionedAt(
    LocalDateTime.now());
    correctionRepository.save(correction);

    // Notify employee
    notificationService.sendNotification(
    correction.getUser(),
    "Correction Approved",
    "Your correction request for "
    + correction.getCorrectionDate()
    + " has been approved by "
    + manager.getFullName(),
    NotificationType.CORRECTION_APPROVED);

    log.info("Correction approved: {} by {}",
    correctionId, managerAdid);

    return "success:Correction approved";
    }

    // ── Manager Dashboard ──────────────────────────────────────

    @GetMapping("/dashboard")
    public String managerDashboard(
    Authentication authentication,
    Model model,
    HttpServletResponse response) {

    response.setHeader(HttpHeaders.CACHE_CONTROL,
    "no-cache, no-store, must-revalidate");

    String adid = authentication.getName();
    loadUserToModel(authentication, model);
    model.addAttribute("activePage", "dashboard");

    // Pending counts for sidebar badges
    long pendingLeaves =
    leaveApprovalService
    .getPendingLeaveCount(adid);
    long pendingCorrections =
    correctionApprovalService
    .getPendingCorrectionCount(adid);

    model.addAttribute("pendingLeaves",
    pendingLeaves);
    model.addAttribute("pendingCorrections",
    pendingCorrections);

    // Today's team attendance
    ListAttendance> teamToday =
    managerService.getTeamAttendance(
    adid, LocalDate.now());

    long presentCount = teamToday.stream()
    .filter(a -> a.getStatus()
    == AttendanceStatus.PRESENT
    || a.getStatus()
    == AttendanceStatus.LATE
    || a.getStatus()
    == AttendanceStatus.HALF_DAY)
    .count();

    long absentCount = teamToday.stream()
    .filter(a -> a.getStatus()
    == AttendanceStatus.ABSENT)
    .count();

    long onLeaveCount = teamToday.stream()
    .filter(a -> a.getStatus()
    == AttendanceStatus.ON_LEAVE)
    .count();

    // Team size
    long teamSize = managerService
    .getTeamSize(adid);

    // Recent pending leaves for dashboard
    ListLeaveRequest> recentLeaves =
      leaveApprovalService
      .getPendingLeaves(adid);

      model.addAttribute("teamToday",    teamToday);
      model.addAttribute("presentCount", presentCount);
      model.addAttribute("absentCount",  absentCount);
      model.addAttribute("onLeaveCount", onLeaveCount);
      model.addAttribute("teamSize",     teamSize);
      model.addAttribute("recentLeaves", recentLeaves);
      model.addAttribute("todayDate",    LocalDate.now());

      return "manager/dashboard";
      }
